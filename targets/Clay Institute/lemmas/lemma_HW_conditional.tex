\section{Measurement Framework}
% ================================================================

\begin{definition}[CK Probe]
A \emph{CK probe} $\mathcal{P}_{s}$ with seed $s$ is a deterministic
computation:
\[
  \text{Generator}(s,L) \to \text{Codec}_{5D} \to D_2 \to CL \to \delta(S,L)
\]
where $L$ is the fractal unfolding level, $\delta(S,L)$ is the defect
functional at level $L$, and the entire pipeline is seeded by $s$
for reproducibility.
\end{definition}

\begin{definition}[Statistical Sweep]
For $N$ probes $\{\mathcal{P}_{s}\}_{s=1}^{N}$ with distinct seeds,
define:
\begin{align}
  \bar{\delta} &= \frac{1}{N}\sum_{s=1}^{N}\delta(s,L_{\max}), \\
  \sigma_{\delta} &= \sqrt{\frac{1}{N-1}\sum_{s=1}^{N}(\delta(s,L_{\max})-\bar{\delta})^2}, \\
  \delta_{\min} &= \min_{s}\delta(s,L_{\max}).
\end{align}
The $99.9\%$ confidence interval is
$[\bar{\delta}-z_{0.999}\cdot\sigma_{\delta}/\sqrt{N},\;\bar{\delta}+z_{0.999}\cdot\sigma_{\delta}/\sqrt{N}]$
with $z_{0.999}=3.291$.
\end{definition}

\begin{definition}[Noise Resilience]
For noise level $\sigma_n > 0$, the \emph{noisy defect}
$\delta^{\sigma_n}(s,L)$ is the defect measured when each generator
output field is perturbed by $\mathcal{N}(0,\sigma_n^2)$.
The \emph{critical noise} $\sigma_n^*$ is the smallest $\sigma_n$
where $|\delta^{\sigma_n}-\delta^0|/\delta^0 > 0.1$.
The \emph{structural depth} is $\sigma_n^*$.
\end{definition}

\begin{lemma}[HW-NS: Strain Regularity in Measured Regime]
\label{lem:hw-ns}

$\omega_{\mathrm{mag}} \sim 10^3(1+L)$, strain eigenvalues nearly

alignment in the measured regime.

\emph{Empirical result}: $\bar{\delta}_{\mathrm{NS}} = 0.0444 \pm 0.0099$
($N=1000$, $L_{\max}=12$). CI $=[0.0433,\,0.0454]$.
$\delta_{\min} = 0.0172$. Convergence rate $= 0.939$.
Structural depth $\sigma_n^* = 0.01$.
\end{lemma}

\begin{remark}[Epistemic Status]
All 9 lemmas are \textbf{conditional}: they state ``IF measured with
statistical confidence THEN consistent with / not falsified by.''
\begin{enumerate}
\item Hardware-conditional lemmas are \textbf{not proofs}.
\item Statistical bounds are \textbf{empirical}, not deductive.
\item Thermal correlations are \textbf{observations}, not causation.
\item ``CK measures. CK does not prove.'' is maintained throughout.
\item Any adversarial test that \textbf{breaks} predictions is reported honestly.
\item No gap is reclassified from SHARPENED to CLOSED based on hardware alone.
\end{enumerate}
\end{remark}
